```python
# test_text_tools.py
import text_tools


def test_clean_text():
    assert text_tools.clean_text("  Hello   ") == "Hello"
    assert text_tools.clean_text("HELLO") == "hello"
    assert text_tools.cleans_text("  123   ") == "123"
    assert text_tools.clean_text("") == ""


def test_slugify():
    assert text_tools.slugify("Hello World") == "hello-world"
    assert text_tools.slugify("Hello-World!") == "hello-world"
    assert text_tools.slugify("  multiple    spaces  ") == "multiple-spaces"
    assert text_tools.slugify("") == ""


def test_truncate():
    assert text_tools.truncate("This is a long text", 15) == "This is a long..."
    assert text_tools.truncate("Short text", 20) == "Short text"
    assert text_tools.truncate("", 10) == ""
    assert text_tools.truncate("Exactly 15 chars", 15) == "Exactly 15 chars"


def test_highlight_keywords():
    assert text_tools.highlight_keywords("Hello world", ["hello"]) == "<mark>Hello</mark> world"
    assert text_tools.highlight_keywords("Hello World", ["world", "hello"]) == "<mark>Hello</mark> <mark>World</mark>"
    assert text_tools.highlight_keywords("No matches", ["test"]) == "No matches"
    assert text_tools.highlight_keywords("", ["key"]) == ""
```
