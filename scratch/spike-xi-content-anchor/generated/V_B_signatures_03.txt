```python
import text_tools


def test_squeeze_runs():
    assert text_tools.squeeze_runs("hello   world") == "hello world"
    assert text_tools.squeeze_runs("a,,b", chars=',') == "a,b"
    assert text_tools.squeeze_runs("  a  b  ", chars=' ') == " a b "
    assert text_tools.squeeze_runs("tab\t\ttab", chars='\t') == "tab\ttab"
    assert text_tools.squeeze_runs("x!!!y", chars='!') == "x!y"


def test_tally_class():
    assert text_tools.tally_class("abc123", "alpha") == 3
    assert text_tools.tally_class("abc123", "digit") == 3
    assert text_tools.tally_class("123", "digit") == 3
    assert text_tools.tally_class("a b c", "space") == 2
    assert text_tools.tally_class("A1b", "alpha") == 2
    assert text_tools.tally_class("123ABC", "digit") == 3
    assert text_tools.tally_class("123ABC", "alpha") == 3
    assert text_tools.tally_class("", "space") == 0
```
