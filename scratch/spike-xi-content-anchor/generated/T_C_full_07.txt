```python
import argparse
import converters

def main():
    parser = argparse.ArgumentParser(description='Convert temperature units.')
    parser.add_argument('value', type=float, help='The temperature value to convert.')
    parser.add_argument('--convert-to', choices=['fahrenheit', 'celsius', 'kelvin'], required=True, help='The target unit.')
    args = parser.parse_args()

    value = args.value
    target = args.convert_to

    if target == 'fahrenheit':
        result = converters.celsius_to_fahrenheit(value)
    elif target == 'celsius':
        result = converters.fahrenheit_to_celsius(value)
    elif target == 'kelvin':
        result = converters.celsius_to_kelvin(value)
    print(result)

if __name__ == '__main__':
    main()
```
