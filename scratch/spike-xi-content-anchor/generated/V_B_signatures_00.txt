```python
import pytest
import text_tools

def test_squeeze_runs_basic():
    assert text_tools.squeeze_runs("hello   world") == "hello world"

def test_squeeze_runs_other_chars():
    assert text_tools.squeeze_runs("a##b", chars="#") == "a#b"

def test_squeeze_runs_no_runs():
    assert text_tools.squeeze_runs("hello world") == "hello world"

def test_squeeze_runs_empty():
    assert text_tools.squeeze_runs("") == ""

def test_squeeze_runs_all_chars():
    assert text_tools.squeeze_runs("#####", chars="#") == "#"

def test_tally_class_digits():
    assert text_tools.tally_class("abc123", predicate_name="digit") == 3

def test_tally_class_alpha():
    assert text_tools.tally_class("abc123", predicate_name="alpha") == 3

def test_tally_class_spaces():
    assert text_tools.tally_class("hello world", predicate_name="space") == 1

def test_tally_class_mixed():
    assert text_tools.tally_class("a1b@C", predicate_name="alpha") == 3

def test_tally_class_empty():
    assert text_tools.tally_class("", predicate_name="digit") == 0

def test_tally_class_all_in_class():
    assert text_tools.tally_class("123", predicate_name="digit") == 3
```
