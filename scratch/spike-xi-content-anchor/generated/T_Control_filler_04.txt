```python
import argparse
from converters import convert

def main() -> None:
    parser = argparse.ArgumentParser(description='Convert temperature between units.')
    parser.add_argument('value', type=float, help='The temperature value to convert.')
    parser.add_argument('from_unit', choices=['C', 'F', 'K'], help='The unit to convert from.')
    parser.add_argument('to_unit', choices=['C', 'F', 'K'], help='The unit to convert to.')
    args = parser.parse_args()

    result = convert(args.value, args.from_unit, args.to_unit)
    print(result)

if __name__ == '__main__':
    main()
```
