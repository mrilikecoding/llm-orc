```python
import argparse
import converters

def main():
    parser = argparse.ArgumentParser(description='Convert temperature')
    parser.add_argument('value', type=float, help='The temperature value')
    parser.add_argument('input_unit', choices=['celsius', 'fahrenheit'], help='Input unit')
    parser.add_argument('target_unit', choices=['fahrenheit', 'celsius', 'kelvin'], help='Target unit')
    args = parser.parse_args()

    if args.input_unit == 'celsius':
        if args.target_unit == 'fahrenheit':
            result = converters.celsius_to_fahrenheit(args.value)
        elif args.target_unit == 'kelvin':
            result = converters.celsius_to_kelvin(args.value)
    elif args.input_unit == 'fahrenheit':
        if args.target_unit == 'celsius':
            result = converters.fahrenheit_to_celsius(args.value)

    print(result)

if __name__ == '__main__':
    main()
```
