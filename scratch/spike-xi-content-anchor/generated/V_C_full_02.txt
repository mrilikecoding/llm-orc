```python
import pytest
from text_tools import squeeze_runs, tally_class

def test_squeeze_runs_normal():
    assert squeeze_runs("a   b  c") == "a b c"

def test_squeeze_runs_other_chars():
    assert squeeze_runs("xxxyyyzzz", chars='x') == "xyyyzzz"

def test_squeeze_runs_multiple_chars():
    assert squeeze_runs("xxyy", chars='xy') == "xy"

def test_squeeze_runs_empty():
    assert squeeze_runs("") == ""

def test_squeeze_runs_all_chars():
    assert squeeze_runs("    ", chars=' ') == " "

def test_tally_class_digit():
    assert tally_class("abc123", "digit") == 3

def test_tally_class_alpha():
    assert tally_class("abc123", "alpha") == 3

def test_tally_class_space():
    assert tally_class("a b c", "space") == 2

def test_tally_class_empty():
    assert tally_class("", "digit") == 0

def test_tally_class_no_match():
    assert tally_class("abc", "digit") == 0
```
