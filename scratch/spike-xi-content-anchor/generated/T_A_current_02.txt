```python
import argparse
import converters

def main():
    parser = argparse.ArgumentParser(description='Convert temperature between units.')
    parser.add_argument('--value', type=float, required=True, help='The temperature value to convert.')
    parser,add_argument('--from', dest='from_unit', required=True, choices=['c', 'f', 'k'], help='The original unit (c, f, k).')
    parser.add_argument('--to', required=True, choices=['c', 'f', 'k'], help='The target unit (c, f, k).')
    args = parser.parse_args()

    result = converters.convert(args.value, args.from_unit, args.to)
    print(result)

if __name__ == '__main__':
    main()
```
