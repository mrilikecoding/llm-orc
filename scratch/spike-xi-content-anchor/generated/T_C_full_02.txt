```python
import argparse
import converters

def main():
    parser = argparse.ArgumentParser(description='Temperature conversion CLI')
    parser.add_argument('--to', required=True, choices=['celsius', 'fahrenheit', 'kelvin'], help='Target unit')
    parser.add_argument('--value', type=float, required=True, help='The value to convert')
    args = parser.parse_args()

    if args.to == 'fahrenheit':
        func = converters.celsius_to_fahrenheit
    elif args.to == 'celsius':
        func = converters.fahrenheit_to_celsius
    elif args.to == 'kelvin':
        func = converters.celsius_to_kelvin
    else:
        parser.error(f"Unsupported target unit: {args.to}")

    result = func(args.value)
    print(result)

if __name__ == '__main__':
    main()
```
