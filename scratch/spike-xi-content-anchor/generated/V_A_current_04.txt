```python
import text_tools

def test_to_snake_case():
    assert text_tools.to_snake_case("HelloWorld") == "hello_world"
    assert text_tools.to_snake_case("helloWorld") == "hello_world"
    assert text_tools.to_snake_case("HELLO") == "hello"

def test_to_camel_case():
    assert text_tools.to_camel_case("hello_world") == "helloworld"
    assert text_tools.to_camel_case("hello world") == "helloworld"
    assert text_tools.to_camel_case("HelloWorld") == "helloworld"

def test_remove_special_characters():
    assert text_tools.remove_special_characters("Hello!@#") == "Hello"
    assert text_tools.remove_special_characters("abc123") == "abc123"
    assert text_tools.remove_special_characters("a_b-c") == "ab_c"
```
