```python
import sys
import converters

def main():
    if len(sys.argv) != 4:
        print("Usage: python cli.py <value> <source_unit> <target_unit>")
        return

    try:
        value = float(sys.argv[1])
        source = sys.argv[2].upper()
        target = sys.argv[3].upper()

        if source == 'C' and target == 'F':
            result = converters.celsius_to_fahrenheit(value)
        elif source == 'F' and target == 'C':
            result = converters.fahrenheit_to_celsius(value)
        else:
            print("Unsupported conversion. Use 'C' for Celsius and 'F' for Fahrenheit.")
            return

        print(f"{value}°{source} is equal to {result}°{target}")

    except ValueError:
        print("Invalid value. Please provide a numeric temperature.")

if __name__ == "__main__":
    main()
```
