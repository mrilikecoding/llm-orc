```python
import argparse
import converters

def main():
    parser = argparse.ArgumentParser(description='Convert temperature.')
    parser.add_argument('value', type=float, help='The input temperature value')
    parser.add_argument('conversion', type=str, help='The conversion type (CtoF, FtoC, CtoK)')
    args = parser.parse_args()

    if args.conversion == 'CtoF':
        result = converters.celsius_to_fahrenheit(args.value)
    elif args.conversion == 'FtoC':
        result = converters.fahrenheit_to_celsius(args.value)
    elif args.conversion == 'CtoK':
        result = converters.celsius_to_kelvin(args.value)
    else:
        parser.error(f"Unknown conversion type: {args.conversion}")

    print(result)

if __name__ == '__main__':
    main()
```
