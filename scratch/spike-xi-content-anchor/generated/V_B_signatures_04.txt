```python
import text_tools

def test_squeeze_runs_default():
    assert text_tools.squeeze_runs("hello   world") == "hello world"

def test_squeeze_runs_custom_chars():
    assert text_tools.squeeze_runs("hello--world", chars="-") == "hello-world"

def test_tally_class_digits():
    assert text_tools.tally_class("abc123", "digit") == 3

def test_tally_class_alpha():
    assert text_tools.tally_class("abc123", "alpha") == 3

def test_tally_class_spaces():
    assert text_tools.tally_class("hello world", "space") == 1

def test_tally_class_empty():
    assert text_tools.tally_class("", "digit") == 0

def test_tally_class_mixed():
    assert text_tools.tally_class("a1 b2 c3", "alpha") == 3
    assert text_tools.tally_class("a1 b2 c3", "digit") == 3
```
